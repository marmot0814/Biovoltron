99  static auto
100  compute_tlen(std::int32_t read_pos, const Cigar& read_cigar,
101  bool read_forward, std::int32_t mate_pos,
102  const Cigar& mate_cigar, bool mate_forward) noexcept
103  -> std::int32_t {
104  if (read_pos > mate_pos)
105  return -compute_tlen(mate_pos, mate_cigar, mate_forward, read_pos,
106  read_cigar, read_forward);
107 
108  switch (const auto read_ori = compute_ori(read_forward, mate_forward);
109  read_ori) {
110  case FR:
111  return mate_pos + mate_cigar.ref_size() - read_pos;
112  case FF:
113  if (const auto tlen = mate_pos + mate_cigar.read_size()
114  - (read_pos + read_cigar.read_size());
115  tlen != 0)
116  return tlen + (tlen > 0 ? 1 : -1);
117  else
118  return 0;
119  case RR:
120  if (const auto tlen = mate_pos + mate_cigar.ref_size()
121  - (read_pos + read_cigar.ref_size());
122  tlen != 0)
123  return tlen + (tlen > 0 ? 1 : -1);
124  else
125  return 0;
126  case RF:
127  if (const auto tlen = mate_pos - (read_pos + read_cigar.ref_size()) + 1;
128  tlen != 0)
129  return tlen + (tlen > 0 ? 1 : -1);
130  else
131  return 0;
132  default:
133  return 0;
134  }
